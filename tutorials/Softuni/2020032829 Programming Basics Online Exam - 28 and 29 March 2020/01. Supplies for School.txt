#include <iostream>
using namespace std;

int main()
{
	int penCount; cin >> penCount;
	int markerCount; cin >> markerCount;
	double cleaningAgent; cin >> cleaningAgent;
	int percent; cin >> percent;

	double totalMoney = 0;
	cout.setf(ios::fixed);
	cout.precision(3);

	totalMoney = ((penCount * 5.8) + (markerCount * 7.2) + (cleaningAgent * 1.2)) * ((100 - percent) / 100.0);
	cout << totalMoney;
	return 0;
}
