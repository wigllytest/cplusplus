#include <iostream>

using namespace std;

int main() {

	int n; cin >> n;

	for (int row = 1; row <= n; row++)
	{
		for (int col = 0; col < n-row; col++)
		{
			cout << " ";
		}
		cout << "*";
		for (int col = 1; col < row; col++)
		{
			cout << " *";
		}
		cout << endl;
	}

	for (int row = 2; row <= n; row++)
	{
		for (int col = 1; col <= row - 1; col++)
		{
			cout << " ";
		}
		cout << "*";
		for (int col = 1; col <= n - row; col++)
		{
			cout << " *";
		}
		cout << endl;

	}
	return 0;
}
