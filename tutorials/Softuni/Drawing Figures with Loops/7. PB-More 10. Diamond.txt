#include <iostream>

using namespace std;

int main() {

	int n; cin >> n;

	int leftRight = (n - 1) / 2;

	for (int i = 1; i <= (n - 1) / 2; i++)
	{
		for (int j = 0; j < leftRight; j++)
		{
			cout << "-";
		}

		cout << "*";

		int mid = n - 2 * leftRight - 2;

		if (mid >= 0)
		{
			for (int j = 0; j < mid; j++)
			{
				cout << "-";
			}

			cout << "*";
		}
		for (int j = 0; j < leftRight; j++)
		{
			cout << "-";
		}
		cout << endl;
		leftRight--;
	}

	for (int i = n / 2; i < n; i++)
	{
		for (int j = 0; j < leftRight; j++)
		{
			cout << "-";
		}
		cout << "*";
		int mid = n - 2 * leftRight - 2;

		if (mid >=0)
		{
			for (int j = 0; j < mid; j++)
			{
				cout << "-";
			}
			cout << "*";
		}	
		for (int j = 0; j < leftRight; j++)
		{
			cout << "-";
		}
		cout << endl;
		leftRight++;
	}

	return 0;
}
