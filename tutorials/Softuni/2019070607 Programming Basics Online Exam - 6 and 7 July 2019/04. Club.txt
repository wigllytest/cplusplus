#include <iostream>
#include <string>

using namespace std;

int main()
{
	double expIncome; cin >> expIncome;
	string cocktail;
	cin.ignore();
	getline(cin, cocktail);
	int cocktailCount;
	
	double income = 0;
	int price = 0;
	double neededMoney = 0;
	while (cocktail != "Party!" || income >= expIncome)
	{
		cin >> cocktailCount;
		price = cocktail.length() * cocktailCount;
		double totalPrice = price;

		if (price % 2 == 1) totalPrice = price * 0.75;

		income += totalPrice;
		if (income >= expIncome) break;

		cin.ignore();
		getline(cin, cocktail);
	}
	cout.setf(ios::fixed);
	cout.precision(2);
	neededMoney = expIncome - income;

	if (cocktail == "Party!")
	{
		cout << "We need " << neededMoney << " leva more." << endl;
	}
	else if (income >= expIncome)
	{
		cout << "Target acquired." << endl;
	}

	cout << "Club income - " << income << " leva.";

	return 0;
}
