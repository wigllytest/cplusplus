#include <iostream>


using namespace std;

int main() {

	int num1, num2;
	char op;
	cin >> num1 >> num2 >> op;

	

	if (op == '+' || op == '-' || op == '*')
	{
		int result = 0;

		if (op == '+')
		{
			result = num1 + num2;
		}
		else if (op == '-')
		{
			result = num1 - num2;
		}
		else if (op == '*')
		{
			result = num1 * num2;
		}

		string numType = "even";
		if (result % 2 != 0 )
		{
			numType = "odd";
		}


		cout << num1 << " " << op << " " << num2 << " = " << result << " - " << numType << endl;

	}
	else if (op == '/')
	{
		cout.setf(ios::fixed);
		cout.precision(2);

		//double result = num1 / num2;
		if (num2 == 0)
		{
			cout << "Cannot divide " << num1 << " by zero";
		}
		else
		{
			cout << num1 << " / " << num2 << " = " << num1 * 1.0 / num2;
		}

	}
	else if (op == '%')
	{	
		if (num2 == 0)
		{
			cout << "Cannot divide " << num1 << " by zero";
		}
		else
		{
			cout << num1 << " % " << num2 << " = " << num1 % num2;
		}
	}

	
	 	
	return 0;
}
