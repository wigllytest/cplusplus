#include <iostream>

using namespace std;

int main() {

	int hrizPcs, rosePcs, tulipsPcs; cin >> hrizPcs >> rosePcs >> tulipsPcs;
	string season, todayIs; cin >> season >> todayIs;
	
	double priceBouquet = 1;

	if (todayIs == "Y") priceBouquet = priceBouquet * 1.15;
	else if (todayIs == "N") priceBouquet = priceBouquet;

	if (season == "Spring" || season == "Summer")
	{	
		if (tulipsPcs > 7 && season == "Spring") priceBouquet = priceBouquet * 0.95 * (hrizPcs *1.0 * 2 + rosePcs * 4.1 + tulipsPcs * 2.5);
		else									 priceBouquet = priceBouquet * (hrizPcs *1.0 * 2 + rosePcs * 4.1 + tulipsPcs * 2.5);
	}
	else if (season == "Autumn" || season == "Winter")
	{
		if (rosePcs >= 10 && season == "Winter") priceBouquet = priceBouquet * 0.9 * (hrizPcs * 3.75 + rosePcs * 4.5 + tulipsPcs * 4.15);
		else									 priceBouquet = priceBouquet * (hrizPcs * 3.75 + rosePcs * 4.5 + tulipsPcs * 4.15);
	}

	if (hrizPcs + rosePcs + tulipsPcs > 20)
	{
		priceBouquet = 0.8 * priceBouquet;
	}

	priceBouquet = priceBouquet + 2;

	cout.setf(ios::fixed);
	cout.precision(2);

	cout << priceBouquet;

	return 0;
}
