#include <iostream>

using namespace std;

int main() {

	int n;
	cin >> n;

	int counter = 0;
	double balance = 0.0;

	cout.setf(ios::fixed);
	cout.precision(2);
	
	while (counter < n)
	{
		double ammount;
		cin >> ammount;

		if (ammount < 0)
		{
			cout << "Invalid operation!" << endl;
			break;
		}

		balance += ammount;
		cout << "Increase: " << ammount << endl;
		counter++;
		
	}



	cout << "Total: " << balance;

	return 0;
}
