#include <iostream>
using namespace std;

int main() {

	int numbCount, numbers; cin >> numbCount;
	int p1Count = 0;
	int p2Count = 0;
	int p3Count = 0;
	double p1 = 0;
	double p2 = 0;
	double p3 = 0;

	for (int i = 0; i < numbCount; i++)
	{
		cin >> numbers;

		if (numbers % 2 == 0) p1Count++;
		if (numbers % 3 == 0) p2Count++;
		if (numbers % 4 == 0) p3Count++;
	}
	cout.setf(ios::fixed);
	cout.precision(2);
	p1 = p1Count * 1.0 / numbCount * 100.0;
	p2 = p2Count * 1.0 / numbCount * 100.0;
	p3 = p3Count * 1.0 / numbCount * 100.0;

	cout << p1 << "%" << endl;
	cout << p2 << "%" << endl;
	cout << p3 << "%" << endl;


	return 0;
}
