#include <iostream>

using namespace std;

int main() {

	int students; cin >> students;
	double grade;
	int gradeTwo = 0;
	int gradeThree = 0;
	int gradeFour = 0;
	int gradeFive = 0;
	double averageResult = 0;

	for (int i = 0; i < students; i++)
	{
		cin >> grade;
		averageResult += grade;
		if      (grade >=2  && grade <3)   gradeTwo++;
		else if (grade >= 3 && grade < 4)  gradeThree++;
		else if (grade >= 4 && grade < 5)  gradeFour++;
		else if (grade >= 5 && grade <= 6) gradeFive++;
	}

	cout.setf(ios::fixed);
	cout.precision(2);

	cout << "Top students: " << gradeFive * 1.0 / students * 100 << "%" << endl;
	cout << "Between 4.00 and 4.99: " << gradeFour * 1.0 / students * 100 << "%" << endl;
	cout << "Between 3.00 and 3.99: " << gradeThree * 1.0 / students * 100 << "%" << endl;
	cout << "Fail: " << gradeTwo * 1.0 / students * 100 << "%" << endl;
	cout << "Average: " << averageResult * 1.0 / students;


	return 0;
}
