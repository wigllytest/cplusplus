#include <iostream>
#include <string>

using namespace std;

int main() {

	int controlNum; cin >> controlNum; // controlNum = a*b + c*d
									   // if a < b && c > d
	int counter = 0;
	string password = "";

	for (int a = 1; a < 10; a++)
	{
		for (int b = 1; b < 10; b++)
		{
			for (int c = 1; c < 10; c++)
			{
				for (int d = 1; d < 10; d++)
				{
					if (controlNum == a * b + c * d)
					{
						if ((a < b) && (c > d))
						{
							counter++;
							
							cout << a << b << c << d << " ";

							if (counter == 4) password = to_string(a) + to_string(b) + to_string(c) + to_string(d);
						}
					}
				}				
			}
		}
	}
	cout << endl;
	if (counter > 3) cout << "Password: " << password;
	else cout << "No!";


	return 0;
}
