#include <iostream>

using namespace std;

int main() {

	int start, end; cin >> start >> end;

	for (int i = start; i <= end; i++)
	{
		for (int j = start; j <= end; j++)
		{
			for (int k = start; k <= end; k++)
			{
				if ((j+k) % 2 == 0 )
				{				
					for (int l = start; l <= end; l++)
					{
						if (i > l)
						{
							if ((i % 2 == 0 && l % 2 == 1) || (i % 2 == 1 && l % 2 == 0))
							{
								cout << i << j << k << l << " ";
							}
						}
						
					}
				}
			}
		}
	}

	return 0;
}
