#include <iostream>
using namespace std;

int main() {

	string figure;
	double squareSide, rectangleLenght, rectangleWidth, circleRadius, triangleside, triangleHeight;
	cin >> figure;

	cout.setf(ios::fixed);
	cout.precision(3);

	if (figure == "square")
	{
		cin >> squareSide;
		cout << squareSide * squareSide;
	}
	else if (figure == "rectangle")
	{
		cin >> rectangleLenght >> rectangleWidth;
		cout << rectangleLenght * rectangleWidth;
	}
	else if (figure == "circle")
	{
		cin >> circleRadius;
		cout << circleRadius * circleRadius * 3.14159;
	}
	else if (figure == "triangle")
	{
		cin >> triangleside >> triangleHeight;
		cout << triangleside * triangleHeight / 2;
	}




	
	return 0;

}
