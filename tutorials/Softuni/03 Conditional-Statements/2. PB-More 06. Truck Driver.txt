#include <iostream>

using namespace std;

int main() {

	string season; cin >> season;
	double kmPerMonth; cin >> kmPerMonth;

	double earnedMoney = 0;
	cout.setf(ios::fixed);
	cout.precision(2);

	if (season == "Spring" || season == "Autumn")
	{ 
		if		(kmPerMonth <= 5000)				         earnedMoney = 4 * kmPerMonth * 0.75;
		else if (kmPerMonth > 5000   && kmPerMonth <= 10000) earnedMoney = 4 * kmPerMonth * 0.95;
		else if (kmPerMonth > 10000  && kmPerMonth <= 20000) earnedMoney = 4 * kmPerMonth * 1.45;
	}
	else if (season == "Summer")
	{

		if		(kmPerMonth <= 5000)				        earnedMoney = 4 * kmPerMonth * 0.9;
		else if (kmPerMonth > 5000  && kmPerMonth <= 10000) earnedMoney = 4 * kmPerMonth * 1.1;
		else if (kmPerMonth > 10000 && kmPerMonth <= 20000) earnedMoney = 4 * kmPerMonth * 1.45;
	}

	else if (season == "Winter")
	{
		if		(kmPerMonth <= 5000)				        earnedMoney = 4 * kmPerMonth * 1.05;
		else if (kmPerMonth > 5000  && kmPerMonth <= 10000) earnedMoney = 4 * kmPerMonth * 1.25;
		else if (kmPerMonth > 10000 && kmPerMonth <= 20000) earnedMoney = 4 * kmPerMonth * 1.45;
	}

	earnedMoney = earnedMoney * 0.9;
	cout << earnedMoney;

	return 0;
}
