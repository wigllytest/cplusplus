#include <iostream>

using namespace std;

int main() {

	string sizeEgg, colorEgg; cin >> sizeEgg >> colorEgg;
	int pieces; cin >> pieces;

	double totalProfit = 0;
	int price = 0;
	if (sizeEgg == "Large")
	{
		if		(colorEgg == "Red")    price = 16;
		else if (colorEgg == "Green")  price = 12;
		else if (colorEgg == "Yellow") price = 9;
	}
	else if (sizeEgg == "Medium")
	{
		if		(colorEgg == "Red")    price = 13;
		else if (colorEgg == "Green")  price = 9;
		else if (colorEgg == "Yellow") price = 7;
	}
	else if (sizeEgg == "Small")
	{
		if		(colorEgg == "Red")    price = 9;
		else if (colorEgg == "Green")  price = 8;
		else if (colorEgg == "Yellow") price = 5;
	}
	cout.setf(ios::fixed);
	cout.precision(2);

	totalProfit = price * 1.0 * pieces * 0.65;

	cout << totalProfit << " leva.";

	return 0;
}
